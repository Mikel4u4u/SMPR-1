  <p>–ê–ª–≥–æ—Ä–∏—Ç–º kNN –æ—Ç–ª–∏—á–∞–µ—Ç—Å—è –æ—Ç 1NN —Ç–µ–º —á—Ç–æ –æ–Ω –æ—Ç–Ω–æ—Å–∏—Ç –∫–ª–∞—Å—Å–∏—Ñ–∏—Ü–∏—Ä—É–µ–º—ã–π —ç–ª–µ–º–µ–Ω—Ç –Ω–µ –∫ –∫–ª–∞—Å—Å—É –±–ª–∏–∂–∞–π—à–µ–≥–æ –∫ –Ω–µ–º—É —ç–ª–µ–º–µ–Ω—Ç–∞, –∞ –∫ –∫–ª–∞—Å—Å—É —á–∞—â–µ –≤—Å–µ–≥–æ –≤—Å—Ç—Ä–µ—á–∞—é—â–µ–º—É—Å—è —Å—Ä–µ–¥–∏ k –±–ª–∏–∂–∞–π—à–∏—Ö —ç–ª–µ–º–µ–Ω—Ç–æ–≤.</p>
  <p>–ü–æ—Å–º–æ—Ç—Ä–∏–º –∫–∞–∫ –æ–Ω–∏ —Ä–∞–±–æ—Ç–∞—é—Ç –Ω–∞ –≤—ã–±–æ—Ä–∫–µ <a href="https://ru.wikipedia.org/wiki/–ò—Ä–∏—Å—ã_–§–∏—à–µ—Ä–∞"; target="_blank">–∏—Ä–∏—Å–æ–≤ —Ñ–∏—à–µ—Ä–∞</a>.</p>
  <p>–î–ª—è –∏–ª–ª—é—Å—Ç—Ä–∞—Ü–∏–∏ –º–µ—Ç–æ–¥–æ–≤, –¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ –ø—Ä–æ–∏–∑–≤–æ–¥–∏—Ç—å –æ–±—É—á–µ–Ω–∏–µ –∏ –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏—é –∏—Å–ø–æ–ª—å–∑—É—è –ª–∏—à—å 2 –ø—Ä–∏–∑–Ω–∞–∫–∞: ¬´–¥–ª–∏–Ω–Ω–∞ –ª–µ–ø–µ—Å—Ç–∫–∞¬ª –∏ ¬´—à–∏—Ä–∏–Ω–∞ –ª–µ–ø–µ—Å—Ç–∫–∞¬ª.</p>
  <p>–õ–∏—Å—Ç–∏–Ω–≥ —Å–∫—Ä–∏–ø—Ç–∞ –Ω–∞ R:</p>
<code>
#–ü–∞–≤–ª—é–∫–æ–≤ –í.–í. 401–∏
<br>#source("script_1NN_kNN.R")
<br>#   –ú–µ—Ç—Ä–∏—á–µ—Å–∫–∏–π –∞–ª–≥–æ—Ä–∏—Ç–º –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏–∏ —Å –æ–±—É—á–∞—é—â–µ–π –≤—ã–±–æ—Ä–∫–æ–π Xl –∏—Å–ø–æ–ª—å–∑—É–µ—Ç
<br>#–í–ï–°–û–í–£–Æ –§–£–ù–ö–¶–ò–Æ w(i,u), –∫–æ—Ç–æ—Ä–∞—è –æ—Ü–µ–Ω–∏–≤–∞–µ—Ç —Å—Ç–µ–ø–µ–Ω—å –≤–∞–∂–Ω–æ—Å—Ç–∏ i-–≥–æ —Å–æ—Å–µ–¥–∞ –¥–ª—è –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏–∏ –æ–±—å–µ–∫—Ç–∞ u –∏ 
<br>#–û–¶–ï–ù–ö–£ –ë–õ–ò–ó–û–°–¢–ò( –ìy(u,Xl) = sum([yui=y]w(i,u)) i –º–µ–Ω—è–µ—Ç—Å—è –æ—Ç 1 –¥–æ l ) –æ–±—å–µ–∫—Ç–∞ u –∫ –∫–ª–∞—Å—Å—É y
<br>#–¥–ª—è —Ç–æ–≥–æ —á—Ç–æ–±—ã –æ—Ç–Ω–µ—Å—Ç–∏ –∫–ª–∞—Å—Å–∏—Ñ–∏—Ü–∏—Ä—É–µ–º—ã–π –æ–±—å–µ–∫—Ç u –∫ —Ç–æ–º—É –∫–ª–∞—Å—Å—É y,
<br>#–¥–ª—è –∫–æ—Ç–æ—Ä–æ–≥–æ —Å—É–º–º–∞—Ä–Ω—ã–π –≤–µ—Å –±–ª–∏–∂–∞–π—à–∏—Ö –æ–±—É—á–∞—é—â–∏—Ö –æ–±—ä–µ–∫—Ç–æ–≤(–æ—Ü–µ–Ω–∫–∞ –±–ª–∏–∑–æ—Å—Ç–∏) –º–∞–∫—Å–∏–º–∞–ª–µ–Ω.
<br>
<br>#–∞–ª–≥–æ—Ä–∏—Ç–º 1NN –æ—Ç–Ω–æ—Å–∏—Ç –∫–ª–∞—Å—Å–∏—Ñ–∏—Ü–∏—Ä—É–µ–º—ã–π –æ–±—å–µ–∫—Ç –∫ —Ç–æ–º—É –∂–µ –∫–ª–∞—Å—Å—É, —á—Ç–æ –∏ –±–ª–∏–∂–∞–π—à–∏–π –æ–±—å–µ–∫—Ç –≤—ã–±–æ—Ä–∫–∏
<br>	#–≤–µ—Å–æ–≤–∞—è —Ñ—É–Ω–∫—Ü–∏—è —Ä–∞–≤–Ω–∞ –µ–¥–∏–Ω–∏—Ü–µ –Ω–∞ –±–ª–∏–∂–∞–π—à–µ–º —ç–ª–µ–º–µ–Ω—Ç–µ –∏ —Ä–∞–≤–Ω–∞ –Ω—É–ª—é –Ω–∞ –æ—Å—Ç–∞–ª—å–Ω—ã—Ö
<br>	#—Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤–µ–Ω–Ω–æ –∏ –æ—Ü–µ–Ω–∫–∞ –±–ª–∏–∑–æ—Å—Ç–∏ —Ä–∞–≤–Ω–∞ 1 –Ω–∞ –±–ª–∏–∂–∞–π—à–µ–º —Å–æ—Å–µ–¥–µ –∏ —Ä–∞–≤–Ω–∞ –Ω—É–ª—é –Ω–∞ –≤—Å–µ—Ö –æ—Å—Ç–∞–ª—å–Ω—ã—Ö
<br>#–∞–ª–≥–æ—Ä–∏—Ç–º kNN –æ—Ç–Ω–æ—Å–∏—Ç –∫–ª–∞—Å—Å–∏—Ñ–∏—Ü–∏—Ä—É–µ–º—ã–π –æ–±—å–µ–∫—Ç –∫ —Ç–æ–º—É –∂–µ –∫–ª–∞—Å—Å—É, –∫ –∫–æ—Ç–æ—Ä–æ–º—É –ø—Ä–∏–Ω–∞–¥–ª–µ–∂–∞—Ç –±–æ–ª—å—à–∏–Ω—Å—Ç–≤–æ –∏–∑ k –±–ª–∏–∂–∞–π—à–∏—Ö –æ–±—å–µ–∫—Ç–æ–≤ –≤—ã–±–æ—Ä–∫–∏
<br>	#–≤–µ—Å–æ–≤–∞—è —Ñ—É–Ω–∫—Ü–∏—è —Ä–∞–≤–Ω–∞ 1 –Ω–∞ k –±–ª–∏–∂–∞–π—à–∏—Ö —Å–æ—Å–µ–¥—è—Ö –∏ 0 –Ω–∞ –æ—Å—Ç–∞–ª—å–Ω—ã—Ö
<br>#–æ–±–∞ –∞–ª–≥–æ—Ä–∏—Ç–º–∞ –æ—Ç–Ω–µ—Å—É—Ç –æ–±—å–µ–∫—Ç –∫ –ø–µ—Ä–≤–æ–º—É –≤—Å—Ç—Ä–µ—Ç–∏–≤—à–µ–º—É—Å—è –≤ –≤—ã–±–æ—Ä–∫–µ –∫–ª–∞—Å—Å—É –≤ —Å–ª—É—á–∞–µ –æ–¥–∏–Ω–∞–∫–æ–≤–æ–≥–æ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—è(1NN)
<br>	–∏–ª–∏ –æ–¥–∏–Ω–∫–æ–≤–æ–≥–æ —á–∏—Å–ª–∞ –æ–±—å–µ–∫—Ç–æ–≤ —Å—Ä–µ–¥–∏ k —Å–æ—Å–µ–¥–µ–π, –ø—Ä–∏–Ω–∞–¥–ª–µ–∂–∞—â–∏—Ö —Ä–∞–∑–Ω—ã–º –∫–ª–∞—Å—Å–∞–º(kNN)
<br>
<br>#–î–ª—è –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏–∏ –Ω—É–∂–Ω—ã:
<br>	#–æ–±—É—á–∞—é—â–∞—è –≤—ã–±–æ—Ä–∫–∞
		Xl = iris#–∏—Ä–∏—Å—ã —Ñ–∏—à–µ—Ä–∞: –æ–±—É—á–∞—é—â–∞—è –≤—ã–±–æ—Ä–∫–∞
		N = c(1,3)#–≤–µ–∫—Ç–æ—Ä –ø—Ä–∏–∑–Ω–∞–∫–æ–≤, —Å–æ–¥–µ—Ä–∂–∏—Ç –Ω–æ–º–µ—Ä–∞ –ø—Ä–∏–∑–Ω–∞–∫–æ–≤ —É—á–∞–≤—Å—Ç–≤—É—é—â–∏—Ö –≤ –æ–±—É—á–µ–Ω–∏–∏
		nomer_stolbtca_classov <- dim(Xl)[2]#–Ω–æ–º–µ—Ä —Å—Ç–æ–ª–±—Ü–∞ —Å –Ω–∞–∑–≤–∞–Ω–∏—è–º–∏ –∫–ª–∞—Å—Å–æ–≤(–≤ –Ω–∞—à–µ–º —Å–ª—É—á–∞–µ –ø–æ—Å–ª–µ–¥–Ω–∏–π 5-–π)
<br>	#—Ñ—É–Ω–∫—Ü–∏—è —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—è
		poschitat_rasstoyanie = function(t1, t2) {#—Ñ—É–Ω–∫—Ü–∏—è —Ä–∞—Å—Å—Ç–æ—è–Ω–∏—è, –≤—ã—á–∏—Å–ª—è–µ—Ç –µ–≤–∫–ª–∏–¥–æ–≤–æ —Ä–∞—Å—Å—Ç–æ—è–Ω–∏–µ –º–µ–∂–¥—É —ç–ª–µ–º–µ–Ω—Ç–∞–º–∏,
			sqrt(sum((t1 - t2)^2))                  #–∫–∞–∫ –±—É–¥—Ç–æ –∑–Ω–∞—á–µ–Ω–∏—è –ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤ —è–≤–ª—è—é—Ç—Å—è –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞–º–∏
		}
<br>	#–º–µ—Ç–æ–¥ —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∏
		otsortirovat_vyborky = function(klassyfitcyruyemyy_element,vyborka,vector_priznakov, po_ubyvaniyu = FALSE, funktciya_vesov = poschitat_rasstoyanie){
<br>		#–≤–æ–∑–≤—Ä–∞—â–∞–µ—Ç –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—É—é –≤—ã–±–æ—Ä–∫—É –ø–æ –≤–æ–∑—Ä–∞—Å—Ç–∞–Ω–∏—é(–ø–æ —É–º–æ–ª—á–∞–Ω–∏—é –ø–æ–¥—Å—Ç–∞–≤–∏–≤ TRUE –æ—Ç—Å–æ—Ä—Ç–∏—Ä—É–µ–º –ø–æ —É–±—ã–≤–∞–Ω–∏—é) –≤–µ—Å–æ–≤(—Ä–∞—Å—Å—Ç–æ—è–Ω–∏–π) –µ–µ —ç–ª–µ–º–µ–Ω—Ç–æ–≤ –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ –∫–ª–∞—Å—Å–∏—Ñ–∏—Ü–∏—Ä—É–µ–º–æ–≥–æ —ç–ª–µ–º–µ–Ω—Ç–∞
			dlinna_vyborki = dim(vyborka)[1]
			matritca_rasstoyaniy = matrix(0,dlinna_vyborki,2)#—Å–æ–¥–µ—Ä–∂–∏—Ç –Ω—É–ª–∏,–Ω—É–∂–Ω–∞ –¥–ª—è —Ö—Ä–∞–Ω–µ–Ω–∏—è –≤–µ—Å–æ–≤ –≤—ã–±–æ—Ä–∫–∏ –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ u
			for (i in 1:dlinna_vyborki) {
				matritca_rasstoyaniy[i,] = c(funktciya_vesov(klassyfitcyruyemyy_element, vyborka[i,][vector_priznakov]), i)
			}
			vyborka[order(matritca_rasstoyaniy[,1],decreasing = po_ubyvaniyu),]
		}
<br>	#–º–µ—Ç–æ–¥—ã –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏–∏:
		NN = function(klassyfitcyruyemyy_element,vyborka, vector_priznakov,nomer_stolbtca_classov,funktciya_rasstoyaniy=poschitat_rasstoyanie) {
			sortirovannaya_vyborka = otsortirovat_vyborky(klassyfitcyruyemyy_element,vyborka,vector_priznakov,FALSE,funktciya_rasstoyaniy)
			sortirovannaya_vyborka[1,nomer_stolbtca_classov]
		}
<br>
		kNN = function(klassyfitcyruyemyy_element,k,vyborka, vector_priznakov,nomer_stolbtca_classov,funktciya_rasstoyaniy=poschitat_rasstoyanie) {
			sortirovannaya_vyborka = otsortirovat_vyborky(klassyfitcyruyemyy_element,vyborka,vector_priznakov,FALSE,funktciya_rasstoyaniy)
			klassy = poluchit_klassy(vyborka,nomer_stolbtca_classov)
			blizost <- rep(0,times = length(klassy))
			names(blizost) <- klassy
			for (i in 1:k) {
				blizost[sortirovannaya_vyborka[i,nomer_stolbtca_classov]] <- (blizost[sortirovannaya_vyborka[i,nomer_stolbtca_classov]] + 1)
			}
			max = 1
			for (i in 1:length(blizost)) {
				if (blizost[max] < blizost[i]) {
				max <- i
				}
			}
			vyborka[vyborka$Species == klassy[max],][1,nomer_stolbtca_classov]#—Ç–∞–∫–∏–µ —Å–ª–æ–∂–Ω–æ—Å—Ç–∏ –¥–ª—è —Ç–æ–≥–æ, —á—Ç–æ–±—ã –≤–æ–∑–≤—Ä–∞—â–∞–ª—Å—è —Ñ—Ä–µ–π–º, –∞ –Ω–µ —Å—Ç—Ä–æ–∫–∞
<br>													#klassy[max] –≤–µ—Ä–Ω–µ—Ç —Å—Ç—Ä–æ–∫—É —Å –Ω–∞–∑–≤–∞–Ω–∏–µ–º –Ω–∞–∏–±–æ–ª–µ–µ —á–∞—Å—Ç–æ –≤—Å—Ç—Ä–µ—á–∞—é—â–µ–≥–æ—Å—è –∫–ª–∞—Å—Å–∞
<br>													#vyborka[vyborka$Species == klassy[max],] —Ñ—Ä–µ–π–º —Å–æ–¥–µ—Ä–∂–∞—â–∏–π —Ç–æ–ª—å–∫–æ 1 –∫–ª–∞—Å—Å
<br>													#vyborka[vyborka$Species == klassy[max],][1,nomer_stolbtca_classov] —Ñ—Ä–µ–π–º —Å–æ–¥–µ—Ä–∂–∞—â–∏–π –Ω–∞–∑–≤–∞–Ω–∏–µ –∫–ª–∞—Å—Å–∞
		}
<br>#–í–∏–∑—É–∞–ª–∏–∑–∞—Ü–∏—è:
	par(mfrow = c(2,1))#   –§—É–Ω–∫—Ü–∏—è par() –∏–∑–º–µ–Ω—è–µ—Ç –ø–æ—Å—Ç–æ—è–Ω–Ω—ã–µ –≥—Ä–∞—Ñ–∏—á–µ—Å–∫–∏–µ –ø–∞—Ä–∞–º–µ—Ç—Ä—ã,
<br>		#—Ç–æ –µ—Å—Ç—å –ø–æ—Å–ª–µ–¥—É—é—â–∏–µ –≥—Ä–∞—Ñ–∏–∫–∏ –±—É–¥—É—Ç —Å—Ç—Ä–æ–∏—Ç—å—Å—è –æ—Ç–Ω–æ—Å–∏—Ç–µ–ª—å–Ω–æ
<br>		#–ø–∞—Ä–∞–º–µ—Ç—Ä–æ–≤, —É–∫–∞–∑–∞–Ω–Ω—ã—Ö –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª–µ–º.mfrow –≤–µ–∫—Ç–æ—Ä —Ñ–æ—Ä–º—ã c (nr, nc), –∫–æ—Ç–æ—Ä—ã–π –¥–µ–ª–∏—Ç –≥—Ä–∞—Ñ–∏—á–µ—Å–∫–æ–µ –æ–∫–Ω–æ
<br>		#–∫–∞–∫ –º–∞—Ç—Ä–∏—Ü—É –∏–∑ nr —Å—Ç—Ä–æ–∫ –∏ nc —Å—Ç–æ–ª–±—Ü–æ–≤, –≥—Ä–∞—Ñ–∏–∫–∏ —Ç–æ–≥–¥–∞ —Ä–∏—Å—É—é—Ç—Å—è –≤ —Å—Ç—Ä–æ–∫–∞—Ö
<br>
	poluchit_klassy = function(vyborka,nomer_stolbtca_classov){#–µ—Å–ª–∏ —è –ø—Ä–∞–≤, —Ç–æ –æ–±—å–µ–¥–∏–Ω–µ–Ω–∏–µ –≤–µ–∫—Ç–æ—Ä–∞ –Ω–∞–∑–≤–∞–Ω–∏–π –∫–ª–∞—Å—Å–æ–≤
<br>															   #—Å —Å–∞–º–∏–º —Å–æ–±–æ–π –¥–æ–ª–∂–Ω–æ —É–¥–∞–ª–∏—Ç—å –æ–¥–∏–Ω–∞–∫–æ–≤—ã–µ —ç–ª–µ–º–µ–Ω—Ç—ã
		union(vyborka[,nomer_stolbtca_classov],vyborka[,nomer_stolbtca_classov])
	}
	klassy <- poluchit_klassy(Xl,nomer_stolbtca_classov) #—Ñ—Ä–µ–π–º —Å–æ–¥–µ—Ä–∂–∞—â–∏–π –Ω–∞–∑–≤–∞–Ω–∏—è –∫–ª–∞—Å—Å–æ–≤
<br>	
	dlinna_vyborki <- dim(Xl)[1]
<br>
	tcveta = topo.colors( length(klassy) )#—Å–æ–∑–¥–∞–µ—Ç –≤–µ–∫—Ç–æ—Ä —Ü–≤–µ—Ç–æ–≤
	points = rbind( #–Ω–µ—Å–∫–æ–ª—å–∫–æ —Ç–æ—á–µ–∫ –¥–ª—è –∫–ª–∞—Å—Å–∏—Ñ–∏–∫–∞—Ü–∏–∏
		c(5, 3),
		c(6.3, 5),
		c(8, 3.5),
		c(5, 2.5),
		c(7.5,2),
		c(6,5.25)
	)
<br>
<br>	# –≥—Ä–∞—Ñ–∏–∫ 1NN
		plot(Xl[,N[1]], Xl[,N[2]], col = tcveta[Xl[,nomer_stolbtca_classov]], xlab = "", ylab = "",main = "1NN")
		for (i in 1:dim(points)[1]) {
			pt = points[i,]
			points(pt[1], pt[2], col = tcveta[NN(pt, Xl, N, nomer_stolbtca_classov )], pch = 15) 
		}
<br>
<br>	# –≥—Ä–∞—Ñ–∏–∫ kNN
		plot(Xl[,N[1]], Xl[,N[2]], col = tcveta[Xl[,nomer_stolbtca_classov]], xlab = "", ylab = "",main = "kNN")
		for (i in 1:dim(points)[1]) {
			pt = points[i,]
			points(pt[1], pt[2], col = tcveta[kNN(pt, 7, Xl, N, nomer_stolbtca_classov )], pch = 15) 
	}
